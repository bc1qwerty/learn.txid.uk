{{- $title := .Get "title" -}} {{- $labels := split (.Get "labels") "," -}} {{- $valuesStr := split (.Get "values") "," -}} {{- $unit := .Get "unit" | default "" -}} {{- /* SVG dimensions */ -}} {{- $w := 600 -}} {{- $h := 200 -}} {{- $pad := 40 -}} {{- $chartW := sub $w (mul $pad 2) -}} {{- $chartH := sub $h (mul $pad 2) -}} {{- /* Calculate min/max */ -}} {{- $maxVal := 0.0 -}} {{- $minVal := 999999.0 -}} {{- range $valuesStr -}} {{- $v := float . -}} {{- if gt $v $maxVal }}{{ $maxVal = $v }}{{ end -}} {{- if lt $v $minVal }}{{ $minVal = $v }}{{ end -}} {{- end -}} {{- if eq $maxVal $minVal }}{{ $maxVal = add $maxVal 1.0 }}{{ end -}} {{- $range := sub $maxVal $minVal -}} {{- /* Build points */ -}} {{- $count := len $labels -}} {{- $points := "" -}} {{- $circles := slice -}} {{- range $i, $_ := $labels -}} {{- $val := index $valuesStr $i | float -}} {{- $x := add $pad (div (mul (float $i) (float $chartW)) (sub (float $count) 1.0)) -}} {{- $y := sub (add $pad (float $chartH)) (div (mul (sub $val $minVal) (float $chartH)) $range) -}} {{- $points = printf "%s%v,%v " $points $x $y -}} {{- $circles = $circles | append (dict "x" $x "y" $y "val" $val "label" (index $labels $i)) -}} {{- end -}}
